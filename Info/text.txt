Individual Disks:
=================
Get-Disk | Where partitionstyle -eq 'raw' | Initialize-Disk -PartitionStyle MBR -PassThru
New-Partition -DiskNumber 2 -DriveLetter F -UseMaximumSize | Format-Volume -FileSystem NTFS -Confirm:$false | Set-Volume -NewFileSystemLabel 'Disk F'
New-Partition -DiskNumber 3 -DriveLetter G -UseMaximumSize | Format-Volume -FileSystem NTFS -Confirm:$false | Set-Volume -NewFileSystemLabel 'Disk G'
New-Partition -DiskNumber 4 -DriveLetter P -UseMaximumSize | Format-Volume -FileSystem NTFS -Confirm:$false | Set-Volume -NewFileSystemLabel 'Pagefile'
New-Partition -DiskNumber 5 -DriveLetter T -UseMaximumSize | Format-Volume -FileSystem NTFS -Confirm:$false | Set-Volume -NewFileSystemLabel 'Temp Disk'

# Initialize Hard Drives
disk_01 = "Get-Disk | Where partitionstyle -eq 'raw' | Initialize-Disk -PartitionStyle MBR -PassThru"

# Configure Disks
disk_02 = "New-Partition -DiskNumber 2 -DriveLetter ${var.disk-f-letter} -UseMaximumSize | Format-Volume -FileSystem NTFS -Confirm:$false | Set-Volume -NewFileSystemLabel '${var.disk-f-label}'"
disk_03 = "New-Partition -DiskNumber 3 -DriveLetter ${var.disk-g-letter} -UseMaximumSize | Format-Volume -FileSystem NTFS -Confirm:$false | Set-Volume -NewFileSystemLabel '${var.disk-g-label}'"

Attach 2 Drives to Available Letters
====================================

Get-Disk | Where partitionstyle -eq 'raw' | Initialize-Disk -PartitionStyle MBR -PassThru | New-Partition -AssignDriveLetter -UseMaximumSize | Format-Volume -FileSystem NTFS -Confirm:$false
Set-Volume -DriveLetter F -NewFileSystemLabel 'Disk F'
Set-Volume -DriveLetter G -NewFileSystemLabel 'Disk G'

# Initialize Hard Drives
disk_01 = "Get-Disk | Where partitionstyle -eq 'raw' | Initialize-Disk -PartitionStyle MBR -PassThru | New-Partition -AssignDriveLetter -UseMaximumSize | Format-Volume -FileSystem NTFS -Confirm:$false"

# Configure Disk Volumen Labels
disk_02 = "Set-Volume -DriveLetter ${var.disk-f-letter} -NewFileSystemLabel '${var.disk-f-label}'"
disk_03 = "Set-Volume -DriveLetter ${var.disk-g-letter} -NewFileSystemLabel '${var.disk-g-label}'"
